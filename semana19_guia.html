💡 Solução Detalhada do Exercício 2

Explicação: Backup automatizado com rsync

📝 Comandos Passo a Passo

#!/bin/bash # backup_completo.sh ORIGEM="/home /etc /var/www" DESTINO="/backup/$(hostname)" DATA=$(date +%Y%m%d) LOG="/var/log/backup.log" mkdir -p $DESTINO echo "[$(date)] Iniciando backup" >> $LOG rsync -avz --delete $ORIGEM $DESTINO/$DATA/ 2>&1 | tee -a $LOG find $DESTINO -type d -mtime +30 -exec rm -rf {} \; 2>/dev/null echo "[$(date)] Backup concluído" >> $LOG # Adicionar ao crontab: # 0 3 * * * /usr/local/bin/backup_completo.sh

📝 Comandos Passo a Passo

# PowerShell: Install-WindowsFeature Windows-Server-Backup $Policy = New-WBPolicy $Volume = Get-WBVolume -VolumePath C: Add-WBVolume -Policy $Policy -Volume $Volume $SystemState = New-WBSystemState Add-WBSystemState -Policy $Policy $BackupLocation = New-WBBackupTarget -VolumePath E: Add-WBBackupTarget -Policy $Policy -Target $BackupLocation $Schedule = New-WBSchedule -DaysOfWeek Sunday,Wednesday -TimesOfDay 02:00 Set-WBSchedule -Policy $Policy -Schedule $Schedule Set-WBPolicy -Policy $Policy Start-WBBackup -Policy $Policy
