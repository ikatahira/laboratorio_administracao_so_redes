📖 GLOSSÁRIO TÉCNICO
Shebang
Primeira linha #!/bin/bash — indica ao sistema operacional qual interpretador executar o script
Exit code
$? — código de retorno do último comando (0 = sucesso, != 0 = erro)
stdin/stdout/stderr
Descritores 0/1/2 — entrada padrão, saída padrão e saída de erros
Subshell
Ambiente filho criado com ( ) ou $( ) que herda variáveis mas não modifica o pai
Quoting
Uso de aspas simples (literal) e duplas (interpola variáveis) para proteger strings
Redirecionamento
>, >>, <, 2>, 2>&1 — redirecionar entradas e saídas de comandos

Primeiro Script: Variáveis e Entrada do Usuário
⏱️ 25 min · Dificuldade: ⭐⭐
+55 XP
▼
🎯 Criar script com shebang, variáveis, read e estrutura básica
📝 CRIANDO O PRIMEIRO SCRIPT
$ nano ~/meu_primeiro.sh

#!/bin/bash
# Meu Primeiro Script Shell
# Autor: Admin | Data: $(date +%Y-%m-%d)

# ── Variáveis ───────────────────────────────
NOME_SERVIDOR="ubuntu-server"
VERSAO="1.0"
DATA=$(date +"%d/%m/%Y %H:%M")    # Captura saída de comando

# ── Saída formatada ─────────────────────────
echo "╔══════════════════════════════╗"
echo "║  Sistema de Boas-vindas v${VERSAO}  ║"
echo "╚══════════════════════════════╝"
echo ""
echo "Servidor : $NOME_SERVIDOR"
echo "Data/Hora: $DATA"
echo "Usuário  : $USER"          # Variável de ambiente
echo "Hostname : $(hostname)"     # Subshell
echo ""

# ── Leitura de entrada ──────────────────────
read -p "Qual é o seu nome? " NOME_USUARIO
read -s -p "Digite uma senha de teste: " SENHA  # -s = oculta digitação
echo ""  # Nova linha após senha oculta

if [ -n "$NOME_USUARIO" ]; then    # -n = string não vazia
    echo "Olá, $NOME_USUARIO! Bem-vindo ao $NOME_SERVIDOR."
else
    echo "Nenhum nome informado."
fi

echo "Senha digitada tem ${#SENHA} caracteres."   # ${#var} = tamanho

$ chmod +x ~/meu_primeiro.sh
$ bash ~/meu_primeiro.sh
📊 VARIÁVEIS ESPECIAIS IMPORTANTES
#!/bin/bash
# Executar como: bash variaveis.sh arg1 arg2 arg3

echo "Script     : $0"    # Nome do script
echo "Argumento 1: $1"    # Primeiro argumento
echo "Argumento 2: $2"
echo "Todos args : $@"    # Todos os argumentos
echo "Total args : $#"    # Quantidade de argumentos
echo "PID atual  : $$"    # PID do script
echo "Exit último: $?"    # Código de retorno

$ bash variaveis.sh hello world 123
⚠️ Aspas: Sempre use aspas duplas ao referenciar variáveis: "$VARIAVEL". Sem aspas, espaços no valor podem quebrar o script.

🔀 SCRIPT COM IF / ELIF / ELSE
#!/bin/bash
# Verificador de sistema

# ── Verificar se é root ─────────────────────
if [ "$EUID" -ne 0 ]; then
    echo "ERRO: Este script requer privilégios root."
    echo "Execute com: sudo $0"
    exit 1
fi

# ── Verificar uso de disco ──────────────────
USO_DISCO=$(df / | tail -1 | awk '{print $5}' | tr -d '%')

if [ "$USO_DISCO" -ge 90 ]; then
    echo "🔴 CRÍTICO: Disco ${USO_DISCO}% cheio! Ação imediata!"
elif [ "$USO_DISCO" -ge 75 ]; then
    echo "🟡 AVISO: Disco ${USO_DISCO}% — monitorar."
elif [ "$USO_DISCO" -ge 50 ]; then
    echo "🟢 OK: Disco ${USO_DISCO}% — normal."
else
    echo "✅ Disco ${USO_DISCO}% — excelente!"
fi

# ── Verificar serviço ───────────────────────
SERVICO="sshd"
if systemctl is-active --quiet "$SERVICO"; then
    echo "✅ $SERVICO está ativo."
else
    echo "❌ $SERVICO está inativo!"
    systemctl start "$SERVICO" && echo "Serviço iniciado."
fi
🔀 CASE — MENU DE OPÇÕES
#!/bin/bash
echo "=== Menu de Administração ==="
echo "1) Status dos serviços"
echo "2) Uso de disco"
echo "3) Usuários logados"
echo "4) Sair"
read -p "Opção: " OPC

case "$OPC" in
    1)
        echo "--- Serviços ativos ---"
        systemctl list-units --type=service --state=active --no-pager | head -15
        ;;
    2)
        echo "--- Uso de disco ---"
        df -h
        ;;
    3)
        echo "--- Usuários logados ---"
        who
        ;;
    4)
        echo "Saindo..."
        exit 0
        ;;
    *)  # Padrão (qualquer outra coisa)
        echo "Opção inválida: $OPC"
        exit 1
        ;;
esac
⚠️ Operadores de comparação: Para números use -eq -ne -lt -gt -le -ge. Para strings use = != -z -n. Não misture: [ "abc" -eq "def" ] causará erro.

Loops: For e While na Prática
⏱️ 25 min · Dificuldade: ⭐⭐⭐
+65 XP

🎯 Usar loops for e while para automatizar tarefas repetitivas
🔁 FOR LOOP — VARIAÇÕES
#!/bin/bash

# ── For em lista ─────────────────────────────
echo "=== Verificando serviços ==="
for SERVICO in ssh apache2 nginx mysql; do
    if systemctl is-active --quiet "$SERVICO" 2>/dev/null; then
        echo "  ✅ $SERVICO: ativo"
    else
        echo "  ❌ $SERVICO: inativo ou não instalado"
    fi
done

# ── For com seq (numérico) ───────────────────
echo ""
echo "=== Criando estrutura de diretórios ==="
for NUM in $(seq 1 5); do
    DIR="/tmp/projeto_${NUM}"
    mkdir -p "$DIR"
    echo "  Criado: $DIR"
done

# ── For em arquivos (glob) ───────────────────
echo ""
echo "=== Scripts em /usr/local/bin ==="
for SCRIPT in /usr/local/bin/*.sh 2>/dev/null; do
    [ -f "$SCRIPT" ] && echo "  $SCRIPT"
done

# ── For C-style (como C/Java) ────────────────
for (( i=0; i<3; i++ )); do
    echo "  Iteração $i"
done
🔁 WHILE LOOP — MONITORAMENTO CONTÍNUO
#!/bin/bash
# Monitorar uso de memória até ficar abaixo do limite
LIMITE=80
INTERVALO=5

echo "Monitorando memória (limite: ${LIMITE}%) — Ctrl+C para sair"

while true; do
    # awk extrai o % de uso de memória
    USO_MEM=$(free | grep Mem | awk '{printf "%.0f", $3/$2*100}')
    DATA=$(date +"%H:%M:%S")

    if [ "$USO_MEM" -gt "$LIMITE" ]; then
        echo "[$DATA] ⚠️  ALERTA: Memória em ${USO_MEM}%!"
    else
        echo "[$DATA] ✅ Memória: ${USO_MEM}%"
    fi

    sleep "$INTERVALO"
done

# Executar em background:
$ bash monitor_mem.sh &
# & = executa em background  |  kill %1 = para o job

#!/bin/bash
# Biblioteca de funções reutilizáveis

# ── Função: log com timestamp ────────────────
log() {
    local NIVEL="$1"    # local = variável local à função
    local MENSAGEM="$2"
    echo "[$(date '+%Y-%m-%d %H:%M:%S')] [$NIVEL] $MENSAGEM" | tee -a /tmp/script.log
}

# ── Função: verifica se comando existe ───────
cmd_existe() {
    command -v "$1" &>/dev/null
    return $?    # retorna exit code do command -v
}

# ── Função: verificar root ───────────────────
verificar_root() {
    if [ "$EUID" -ne 0 ]; then
        log "ERRO" "Necessário executar como root"
        exit 1
    fi
}

# ── Chamadas ─────────────────────────────────
log "INFO" "Script iniciado"
log "WARN" "Espaço em disco baixo"

if cmd_existe "rsync"; then
    log "INFO" "rsync disponível"
else
    log "ERRO" "rsync não encontrado — instale com: apt install rsync"
fi
💾 SCRIPT DE BACKUP COMPLETO
#!/bin/bash
# backup.sh — Script de backup com funções
# Uso: sudo bash backup.sh [diretório_origem]

set -euo pipefail    # e=exit on error u=undef error o=pipe error

# ── Configurações ────────────────────────────
ORIGEM="${1:-/etc}"              # Padrão: /etc
DESTINO="/backup"
DATA=$(date +"%Y%m%d_%H%M%S")
ARQUIVO="backup_${DATA}.tar.gz"
LOG="/var/log/backup.log"

# ── Função de log ─────────────────────────────
log() { echo "[$(date '+%H:%M:%S')] $1" | tee -a "$LOG"; }

# ── Verificações ─────────────────────────────
[ ! -d "$ORIGEM" ] && { log "ERRO: $ORIGEM não encontrado"; exit 1; }

mkdir -p "$DESTINO"
log "=== Backup iniciado ==="
log "Origem : $ORIGEM"
log "Destino: $DESTINO/$ARQUIVO"

# ── Criar backup ─────────────────────────────
if tar -czf "$DESTINO/$ARQUIVO" "$ORIGEM" 2>/dev/null; then
    TAMANHO=$(du -sh "$DESTINO/$ARQUIVO" | cut -f1)
    log "✅ Backup criado: $ARQUIVO ($TAMANHO)"
else
    log "❌ ERRO ao criar backup!"
    exit 1
fi

# ── Rotação: manter apenas últimos 7 backups ─
TOTAL=$(ls -1 "$DESTINO"/backup_*.tar.gz 2>/dev/null | wc -l)
if [ "$TOTAL" -gt 7 ]; then
    ls -1t "$DESTINO"/backup_*.tar.gz | tail -n +8 | xargs rm -f
    log "Rotação: removidos backups antigos (mantidos 7)"
fi

log "=== Backup concluído ==="

$ chmod +x backup.sh && sudo bash backup.sh /home
✅ Boas práticas em scripts: use set -euo pipefail no início · comente o código · use funções para código repetido · sempre valide entradas · log todas as operações importantes
