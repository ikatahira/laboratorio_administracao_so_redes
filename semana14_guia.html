Explicação: Criação de usuários do Active Directory

📝 Comandos Passo a Passo

# PowerShell: New-ADUser -Name "João Silva" -GivenName "João" -Surname "Silva" -SamAccountName "jsilva" -UserPrincipalName "jsilva@empresa.local" -Path "OU=TI,OU=Usuarios,OU=Empresa,DC=empresa,DC=local" -AccountPassword (ConvertTo-SecureString 'P@ssw0rd123' -AsPlainText -Force) -Enabled $true New-ADUser -Name "Maria Santos" -GivenName "Maria" -Surname "Santos" -SamAccountName "msantos" -UserPrincipalName "msantos@empresa.local" -Path "OU=RH,OU=Usuarios,OU=Empresa,DC=empresa,DC=local" -AccountPassword (ConvertTo-SecureString 'P@ssw0rd123' -AsPlainText -Force) -Enabled $true

📝 Comandos Passo a Passo

# PowerShell: New-ADGroup -Name "GRP_TI" -GroupScope Global -GroupCategory Security -Path "OU=Grupos,OU=Empresa,DC=empresa,DC=local" New-ADGroup -Name "GRP_RH" -GroupScope Global -GroupCategory Security -Path "OU=Grupos,OU=Empresa,DC=empresa,DC=local" Add-ADGroupMember -Identity "GRP_TI" -Members "jsilva" Add-ADGroupMember -Identity "GRP_RH" -Members "msantos" New-ADGroup -Name "GRP_Administradores" -GroupScope Universal -GroupCategory Security -Path "OU=Grupos,OU=Empresa,DC=empresa,DC=local" Add-ADGroupMember -Identity "GRP_Administradores" -Members "GRP_TI"
